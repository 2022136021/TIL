import 'package:flutter/material.dart';

void main() {
  runApp(const MyApp());
}

class MyApp extends StatelessWidget {
  const MyApp({super.key});

  // This widget is the root of your application.
  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Flutter Demo',
      theme: ThemeData(
        colorScheme: ColorScheme.fromSeed(seedColor: const Color.fromARGB(255, 230, 115, 111)),
      ),
      home: const MyHomePage(title: '플러터 데모 페이지'),
    );
  }
}

class MyHomePage extends StatefulWidget {
  const MyHomePage({super.key, required this.title});

  final String title;

  @override
  State<MyHomePage> createState() => _MyHomePageState();
}

class _MyHomePageState extends State<MyHomePage> {
  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        backgroundColor: Theme.of(context).colorScheme.inversePrimary,
        title: Text(widget.title),
      ),
      body: Center(
        child: createBody5(),
      ),
    );
  }
}

Widget createBody1() {
  return Container(
    padding: const EdgeInsets.only(
    left: 20,
    right: 20,
    ),
    width: 200,
    height: 100,
    color: Colors.red,
    alignment: Alignment.center,
    child: Text('Container'),
  );
}

Widget createBody2() {
  return Container(
    padding: const EdgeInsets.only(
    left: 20,
    right: 20,
  ),
  decoration: BoxDecoration(
    gradient: LinearGradient(
      colors: [
        Color.fromARGB(255, 255, 59, 98).withOpacity(0.7),
          Color.fromARGB(255, 255, 59, 98)
      ],
      begin: Alignment.topLeft,
      end: Alignment.bottomRight,
       ),
       borderRadius: BorderRadius.circular(10),
       boxShadow: [
        BoxShadow(
          color:Color.fromARGB(255, 255, 59, 98).withOpacity(0.5),
          spreadRadius: 5,
          blurRadius: 7,
          offset: Offset(0, 3),
        ),
       ],
  ),
  width: 200,
  height: 150,
  child: Center(child: Text(
    'Container',
    style:TextStyle(color: Colors.white),
    ),
    ),
  );
}
